#!/usr/bin/perl

use LocalDepth;

use Gopts ; eval Gopts::set_opts         ;
$ob = new LocalDepth                     ;

$ob -> db($opt_db)                       ;

$ob -> write ;
$ob -> fmt                               ;

my @depth_arr = split (' ', $opt_depth ) ;

my $struct = $ob -> depth(@depth_arr)        ;

print $ob -> dump( eval $struct )        ;

$ob -> DESTROY                           ; 

exit
